```{python}
import sqlite3
import pandas as pd
import seaborn as sns
import os

# palmerpenguinsデータセットを読み込む
penguins = sns.load_dataset('penguins')

# 欠損値を削除する
penguins = penguins.dropna()

# SQLiteデータベースに接続（データベースが存在しない場合は新規作成）
conn = sqlite3.connect(os.path.join(r.cur_dir, 'penguins.db'))
cursor = conn.cursor()

# テーブルを作成
cursor.execute('''
    CREATE TABLE IF NOT EXISTS penguins (
        species TEXT,
        island TEXT,
        bill_length_mm REAL,
        bill_depth_mm REAL,
        flipper_length_mm REAL,
        body_mass_g REAL,
        sex TEXT
    )
''')

cursor.execute("DELETE FROM penguins")

# pandas DataFrameをSQLiteテーブルに書き込む
penguins.to_sql('penguins', conn, if_exists='replace', index=False)

# 接続をコミットして閉じる
conn.commit()
conn.close()


if os.path.exists(os.path.join(r.cur_dir, 'penguins.db')):
    print("SQLiteデータベース 'penguins.db' が既に存在します。")
else:
  print("penguins.dbの作成に失敗しました")

```

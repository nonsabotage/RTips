1.3 ステップ2:検証ルールの作成

Code 
agent <- 

  agent |>
  
  # 値がある
  col_exists(columns = c(bill_length_mm)) |> 

  # 数値型
  col_is_numeric(columns = vars(bill_length_mm)) |> 

  # NULLを含まない
  col_vals_not_null(columns = vars(species)) |>

  # 特定の値の範囲
  col_vals_between(columns = vars(bill_length_mm), left = 30, right = 60) |>

  # 特定の水準の範囲
  col_vals_in_set(columns = vars(island), set = c("Biscoe", "Dream", "Torgersen")) |>

  # ある値より大きい
  col_vals_gte(columns = vars(body_mass_g), 0) |> 

  # 条件付きバリデーション
  col_vals_gte(
    columns = vars(bill_length_mm), 
    value = 35, 
    preconditions = ~ . %>% filter(species == "Adelie")

  )

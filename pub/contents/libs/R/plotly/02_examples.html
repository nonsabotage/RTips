1.2 散布図

Code 
library(plotly)

# サンプルデータ
set.seed(123)
x <- rnorm(100)
y <- rnorm(100)
z <- rnorm(100)
category <- sample(c("A", "B", "C"), 100, replace = TRUE)

# 散布図の作成
p <- plot_ly(
  x = x,
  y = y,
  mode = 'markers',  # 散布図をマーカーとして表示
  type = 'scatter',
  marker = list(
    color = z,            # マーカーの色をzに基づいて設定
    size = 10,            # マーカーのサイズ
    opacity = 0.7,        # マーカーの透明度
    colorbar = list(title = "Z値")  # カラーバーのタイトル
  ),
  text = paste("Category: ", category),  # ツールチップに表示するテキスト
  hoverinfo = 'text'  # ツールチップの表示情報
) %>%
  layout(
    title = list(text="散布図のカスタマイズ例", size=40),  # タイトル
    xaxis = list(
      title = "X軸",              # X軸のラベル
      titlefont = list(size = 18),  # X軸ラベルのフォントサイズ
      tickangle = 45,             # X軸の目盛りラベルを45度回転
      showgrid = TRUE,            # X軸のグリッド線を表示
      gridcolor = 'lightgray',    # グリッド線の色
      zeroline = FALSE            # X軸のゼロラインを非表示
    ),
    yaxis = list(
      title = "Y軸",              # Y軸のラベル
      titlefont = list(size = 18),  # Y軸ラベルのフォントサイズ
      showgrid = TRUE,            # Y軸のグリッド線を表示
      gridcolor = 'lightgray',    # グリッド線の色
      zeroline = FALSE            # Y軸のゼロラインを非表示
    ),
    showlegend = TRUE,             # 凡例を表示
    legend = list(
      title = 'カテゴリー',      # 凡例のタイトル
      x = 0.8, y = 0.95,         # 凡例の位置（相対座標）
      bgcolor = 'rgba(255, 255, 255, 0.5)',  # 凡例の背景色
      bordercolor = 'black',     # 凡例のボーダー色
      borderwidth = 1            # 凡例のボーダー幅
    ),
    plot_bgcolor = 'lightgray',   # プロット領域の背景色
    paper_bgcolor = 'white',      # 全体の背景色
    margin = list(l = 60, r = 60, t = 60, b = 60)  # マージンの調整
  )

# プロットの表示

1.3 横棒グラフ

Code 
# 必要なライブラリをインストール＆読み込み
# install.packages("plotly") # 初回のみ必要
library(plotly)

# データの定義
data <- data.frame(
  Category = c("A", "B", "C", "D", "E"),
  Value = c(30, 20, 50, 40, 60)
)

# 横棒グラフの作成
plot <- plot_ly(
  data = data,
  x = ~Value,                # 横軸のデータ
  y = ~Category,             # 縦軸のデータ
  type = 'bar',              # 棒グラフを指定
  orientation = 'h',         # 横棒グラフにする
  marker = list(color = 'rgba(255,99,71,0.6)') # 色の設定
) %>%
  # ツールチップとホバー情報のカスタマイズ
  add_trace(
    text = ~paste("カテゴリ:", Category, "<br>値:", Value),
    hoverinfo = "text"
  ) %>%
  # レイアウトの設定
  layout(
    title = "横棒グラフの例",             # グラフのタイトル
    xaxis = list(title = "値"),          # 横軸のラベル
    yaxis = list(title = "カテゴリ"),    # 縦軸のラベル
    margin = list(l = 100, t = 100, b = 100),              # 左のマージンを調整
    barmode = 'stack',                   # スタック表示（必要に応じて変更可能）
    showlegend = TRUE                    # 凡例を表示
  ) %>%
  # アニメーション設定（必要に応じて）
  animation_opts(
    frame = 100, 
    easing = 'linear', 
    redraw = FALSE
  )

# プロットを表示

1.4 色を変える棒グラフ

Code 
library(plotly)

df <- data.frame(
  Category = c("A", "B", "C", "D", "E"),
  Value = c(30, 20, 50, 40, 60)
)

# "C"だけを別トレースとして分ける
df_highlight <- subset(df, Category == "C")
df_others    <- subset(df, Category != "C")

plot_ly() %>%
  # Cの部分だけのトレース
  add_trace(
    data = df_highlight,
    x = ~Category,
    y = ~Value,
    type = "bar",
    marker = list(color = "red"),
    name = "特別な棒"
  ) %>%
  # それ以外の部分のトレース
  add_trace(
    data = df_others,
    x = ~Category,
    y = ~Value,
    type = "bar",
    marker = list(color = "blue"),
    name = "その他の棒"
  ) %>%
  layout(
    title = "複数トレースで色を変える例",
    xaxis = list(title = "Category"),
    yaxis = list(title = "Value"),
    barmode = "group"  # 棒を横並びに表示（必要に応じて "stack" など）
  )

1.5 ヒートマップ

Code 
# ライブラリを読み込み（必要に応じてインストール）
# install.packages("plotly")
library(plotly)

# サンプルとなる行列データを作成
set.seed(123)  # 乱数シードで再現性を確保
data_matrix <- matrix(
  rnorm(100, mean = 5, sd = 2),  # 平均5、標準偏差2の正規乱数を100個生成
  nrow = 10,                     # 行数
  ncol = 10                      # 列数
)

# 行名と列名を設定
rownames(data_matrix) <- paste0("Row", 1:10)
colnames(data_matrix) <- paste0("Col", 1:10)

# ヒートマップの作成
p <- plot_ly(
  x = colnames(data_matrix),       # x軸ラベルに列名を使用
  y = rownames(data_matrix),       # y軸ラベルに行名を使用
  z = data_matrix,                 # ヒートマップの値
  type = "heatmap",                # ヒートマップを指定
  colorscale = "Viridis",          # カラースケール（"Blues", "Reds"なども可）
  hovertemplate = paste0(
    "行: %{y}<br>",
    "列: %{x}<br>",
    "値: %{z:.2f}<extra></extra>"
  )
  # hovertemplate で表示内容を自由にカスタマイズ
  # %{z:.2f} として小数点以下2桁まで表示
) %>%
  # カラーバー（凡例）の設定
  colorbar(
    title = "値",  # カラーバーのタイトル
    len = 0.8,     # カラーバーの長さ（全体比）
    x = 1.05       # カラーバーをグラフの右側にオフセット
  ) %>%
  # レイアウトの設定
  layout(
    title = "ヒートマップの例",
    xaxis = list(
      title = "列",
      tickangle = 45  # x軸ラベルを45度回転
    ),
    yaxis = list(
      title = "行",
      autorange = "reversed"  # 通常のイメージに合わせ、y軸を上から下へ
    ),
    margin = list(
      b = 100,  # 下マージン
      l = 100,  # 左マージン
      t = 50,   # 上マージン
      r = 120   # 右マージン（カラーバーと被らないように調整）
    ),
    plot_bgcolor = "#f0f0f0"  # 背景色の設定, 
    ,   hoverlabel = list(
      bgcolor = "white",           # 背景色
      bordercolor = "black",       # 枠線の色
      font = list(size = 36),      # フォント設定
      align = "left",              # テキストの配置（"left", "right", "auto"）
      namelength = -1,             # Hover時のトレース名省略をしない
      pad = list(l = 124, r = 10, t = 15, b = 15)  # パディングを設定
    )
  )

# グラフを表示

# 必要に応じてインストール
# install.packages("plotly")
library(plotly)

# サンプルデータ: 日本の主な都市と人口（ざっくりした数字）
df <- data.frame(
  city       = c("Tokyo", "Osaka", "Kyoto", "Sapporo", "Fukuoka"),
  latitude   = c(35.6895, 34.6937, 35.0116, 43.0642, 33.5904),
  longitude  = c(139.6917, 135.5022, 135.7681, 141.3468, 130.4017),
  population = c(14, 2.7, 1.5, 1.9, 1.6)  # 単位: 百万人
)

fig <- plot_ly() %>%
  add_trace(
    type = "scattergeo",
    mode = "markers",
    lat = ~df$latitude,
    lon = ~df$longitude,
    text = ~paste0(
      df$city,
      "<br>人口: ",
      df$population, " 百万人"
    ),
    hoverinfo = "text",
    marker = list(
      size = ~df$population * 5,  # 人口に応じた点の大きさ
      color = "red",
      line = list(width = 1, color = "black")
    )
  ) %>%
  layout(
    title = "日本主要都市の人口",
    geo = list(
      scope = "world",         # "asia" を指定するとアジア中心など範囲を限定可
      resolution = 50,         # 地図の解像度
      showland = TRUE,         # 陸地の表示
      landcolor = "rgb(243, 243, 243)",
      countrycolor = "rgb(204, 204, 204)"
    )
  )

1.9 subplot

Code 
# 必要に応じてインストール
# install.packages("plotly")
library(plotly)

# 1. データ準備
set.seed(123)
n <- 200
x <- rnorm(n, mean = 50, sd = 10)
y <- rnorm(n, mean = 30, sd =  5)

# 2. 中央の散布図
p_scatter <- plot_ly() %>%
  add_markers(
    x = ~x,
    y = ~y,
    marker = list(color = "blue", opacity = 0.6),
    name = "Scatter"
  ) %>%
  layout(
    xaxis = list(showgrid = FALSE),
    yaxis = list(showgrid = FALSE)
  )

# 3. 上側のヒストグラム（x軸の分布）
p_xhist <- plot_ly() %>%
  add_histogram(
    x = ~x,
    marker = list(color = "blue", opacity = 0.6),
    showlegend = FALSE
  ) %>%
  layout(
    yaxis = list(showticklabels = FALSE),  # y軸目盛りラベルを消す
    margin = list(b = 20)                  # 下マージンを少し空ける
  )

# 4. 左側のヒストグラム（y軸の分布）
p_yhist <- plot_ly() %>%
  add_histogram(
    y = ~y,
    marker = list(color = "blue", opacity = 0.6),
    showlegend = FALSE
  ) %>%
  layout(
    xaxis = list(showticklabels = FALSE),  # x軸目盛りラベルを消す
    margin = list(l = 20)                  # 左マージンを少し空ける
  )

p_empty <- plotly_empty(type = "scatter", mode = "markers") %>%
  layout(
    xaxis = list(showgrid = FALSE, zeroline = FALSE, visible = FALSE),
    yaxis = list(showgrid = FALSE, zeroline = FALSE, visible = FALSE)
  )

# 5. subplot() でレイアウトを組み合わせる
#    2×2グリッド: 上左(xヒスト), 上右(空), 下左(yヒスト), 下右(散布図)
fig <- subplot(
  p_xhist,      p_empty,
   p_scatter, p_yhist,
  nrows = 2,
  shareX = FALSE,    # x軸を共有
  shareY = FALSE,    # y軸を共有
  widths = c(0.7, 0.3),   # 2列の幅の比率
  heights = c(0.3, 0.7)   # 2行の高さの比率
)

# 6. 追加レイアウト調整（マージンや軸ラベルなど）
fig <- fig %>%
  layout(
    showlegend = FALSE,
    margin = list(l = 50, r = 50, t = 30, b = 50),
    # 軸の domain を指定すると、位置の微調整ができる
    xaxis = list(domain = c(0.0, 0.7)),    # 下段左領域
    xaxis2 = list(domain = c(0.0, 0.7)),   # 上段左領域 (p_xhist)
    yaxis = list(domain = c(0.3, 1.0)),    # 上段
    yaxis2 = list(domain = c(0.0, 0.3), side = "right"), # 下段
    xaxis2 = list(title = "X values"),
    yaxis2 = list(title = "Y values")
  )

# 7. プロットを表示

```{r}
library(plotly)

# サンプルデータ
x <- c(1, 2, 3, 4, 5)
y <- c(10, 11, 12, 13, 14)
y2 <- c(14, 13, 12, 11, 10)
```

```{r}
# 必要に応じてインストール
# install.packages("plotly")
library(plotly)

# 1. データ準備
set.seed(123)
n <- 200
x <- rnorm(n, mean = 50, sd = 10)
y <- rnorm(n, mean = 30, sd =  5)

# 2. 中央の散布図
p_scatter <- plot_ly() %>%
  add_markers(
    x = ~x,
    y = ~y,
    marker = list(color = "blue", opacity = 0.6),
    name = "Scatter"
  ) %>%
  layout(
    xaxis = list(showgrid = FALSE),
    yaxis = list(showgrid = FALSE)
  )

# 3. 上側のヒストグラム（x軸の分布）
p_xhist <- plot_ly() %>%
  add_histogram(
    x = ~x,
    marker = list(color = "blue", opacity = 0.6),
    showlegend = FALSE
  ) %>%
  layout(
    yaxis = list(showticklabels = FALSE),  # y軸目盛りラベルを消す
    margin = list(b = 20)                  # 下マージンを少し空ける
  )

# 4. 左側のヒストグラム（y軸の分布）
p_yhist <- plot_ly() %>%
  add_histogram(
    y = ~y,
    marker = list(color = "blue", opacity = 0.6),
    showlegend = FALSE
  ) %>%
  layout(
    xaxis = list(showticklabels = FALSE),  # x軸目盛りラベルを消す
    margin = list(l = 20)                  # 左マージンを少し空ける
  )

p_empty <- plotly_empty(type = "scatter", mode = "markers") %>%
  layout(
    xaxis = list(showgrid = FALSE, zeroline = FALSE, visible = FALSE),
    yaxis = list(showgrid = FALSE, zeroline = FALSE, visible = FALSE)
  )

# 5. subplot() でレイアウトを組み合わせる
#    2×2グリッド: 上左(xヒスト), 上右(空), 下左(yヒスト), 下右(散布図)
fig <- subplot(
  p_xhist,      p_empty,
   p_scatter, p_yhist,
  nrows = 2,
  shareX = FALSE,    # x軸を共有
  shareY = FALSE,    # y軸を共有
  widths = c(0.7, 0.3),   # 2列の幅の比率
  heights = c(0.3, 0.7)   # 2行の高さの比率
)

# 6. 追加レイアウト調整（マージンや軸ラベルなど）
fig <- fig %>%
  layout(
    showlegend = FALSE,
    margin = list(l = 50, r = 50, t = 30, b = 50),
    # 軸の domain を指定すると、位置の微調整ができる
    xaxis = list(domain = c(0.0, 0.7)),    # 下段左領域
    xaxis2 = list(domain = c(0.0, 0.7)),   # 上段左領域 (p_xhist)
    yaxis = list(domain = c(0.3, 1.0)),    # 上段
    yaxis2 = list(domain = c(0.0, 0.3), side = "right"), # 下段
    xaxis2 = list(title = "X values"),
    yaxis2 = list(title = "Y values")
  )

# 7. プロットを表示
fig
```

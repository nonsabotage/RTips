1 Create a graph

Code 
g <- make_empty_graph()

g
#> IGRAPH a16e811 D--- 0 0 -- 
#> + edges from a16e811:
  • make_graph系はエッジからグラフを構築する
  • 10個の頂点と2のエッジからなるグラフは次である
    • edgeは1to2, 1to5の2つである
Code 
g <- make_graph(
  edges = c(1, 2,
            1, 5), 
  n = 10, 
  directed = FALSE
)
g
#> IGRAPH a178bde U--- 10 2 -- 
#> + edges from a178bde:
#> [1] 1--2 1--5
Code 
plot(g)

特殊な式でエッジを指定することも可能である。 ~からはじめることで次のようにエッジを指定できる

Code 
# ノードはエッジと重複していても大丈夫である
make_graph(
  ~ 1--2, 1--5, 3, 4, 5
) |> plot()

グラフの名前を使ってグラフを構築することも可能である。 たとえば、Zacharyというグラフは次のように構築できる

Code 
make_graph("Zachary") |> plot()

より詳しい詳細はこの後のチュートリアルに追記される。

2 Vertex and edge IDs

igraphの頂点とエッジはそれぞれ数値IDを有してる。 頂点のIDは必ず1からはじまる。また数値IDは連続しています。 たとえばサブグラフが構築されたときには、 サブグラフの中で頂点番号が振り直されます。 *Pythonとかは0からはじまるので注意すること。

3 Adding/deleting vertices and edges

Code 
g <- make_empty_graph()
g <- add_vertices(g, 3) # 頂点を3つ追加する
plot(g)

もう一度頂点を追加してみる。頂点が３つ追加されて、６つになっていることがわかる。

Code 
g <- add_vertices(g, 3)
plot(g)

次はエッジを追加する。存在しない頂点番号を指定するとエラーになる。

Code 
g <- add_edges(g, edges = c(1, 3, 1, 6, 3, 4))
plot(g)

次の記法でも追加することが出来る。 ただしadd_edgesを使う方が効率的である。

Code 
g + edges(c(2, 5))
#> IGRAPH a1dfc57 D--- 6 4 -- 
#> + edges from a1dfc57:
#> [1] 1->3 1->6 3->4 2->5

より複雑な追加をしてみる.

Code 
g <- 
  make_graph("Zachary") |> 
  add_edges(edges = c(1, 34)) |> 
  add_vertices(6) |> 
  add_edges(edges = c(38, 39, 39, 40, 40, 38, 40, 37))

g

#> + ... omitted several edges

VertexのIDは常に1からはじまり連続的である。 Vertexがrenumberedされたとき、エッジはrenumberedされない。 しかし、エッジの頂点はrenumberedされる。

delete_verticesとdelete_edgesを使い操作してみる。

Code 
# エッジ番号
get.edge.ids(g, c(1, 34))
#> [1] 79
Code 
# delete
g <- delete_edges(g, 79)
get.edge.ids(g, c(1, 34)) # return 0
#> [1] 0

例題として次のグラフを考える。リングを作成して、1と10のエッジを削除する。

Code 
g <- make_ring(10) %>% delete_edges("10|1")
plot(g)

同様に色々試す。

Code 
g <- make_ring(5)
g <- delete_edges(g, get.edge.ids(g, c(1,5, 4,5)))
plot(g)

リテラル表現のグラフ構造である。

Code 
g1 <- graph_from_literal(A-B:C:I, B-A:C:D, C-A:B:E:H, D-B:E:F,
                         E-C:D:F:H, F-D:E:G, G-F:H, H-C:E:G:I,
                         I-A:H)
plot(g1)

グラフがchordalであるのかを判定し、足りないエッジを調べる。

Code 
is_chordal(g1, fillin=TRUE)
#> $chordal
#> [1] FALSE
#> 
#> $fillin
#>  [1] 2 6 8 7 5 7 2 7 6 1 7 1
#> 
#> $newgraph
#> NULL

足りないエッジを追加する。

Code 
chordal_graph <- add_edges(g1, is_chordal(g1, fillin=TRUE)$fillin)
plot(chordal_graph)

4 Constucting graphs

igraphにはここまで使ったmake_graph系以外にも多くのグラフ構築関数がある。 このグラフは127のノードを持ち、いずれノードも2つのエッジを持つ。

Code 
graph1 <- make_tree(127, 2, mode = "undirected")
summary(graph1)
#> IGRAPH a2400a3 U--- 127 126 -- Tree
#> + attr: name (g/c), children (g/n), mode (g/c)
Code 
plot(graph1)

同じグラフを作成して、比較をしてみる。

Code 
graph2 <- make_tree(127, 2, mode = "undirected")
identical_graphs(graph1, graph2)
#> [1] TRUE

GRGsで同じパラメータを使うと異なる構造のグラフが生成される。

Code 
graph2 <- sample_grg(100, .2)
identical_graphs(graph1, graph2)
#> [1] FALSE

２つのグラフが同型であるか、つまり頂点と辺の数が同じで1対1に対応させられる場合である。

Code 
isomorphic(graph1, graph2)
#> [1] FALSE

5 Setting and retrieving graph

VertexやEdgesはIDだけでなく属性を付与することができる。

Code 
g <- make_graph(~ Alice-Bob:Claire:Frank, Claire-Alice:Dennis:Frank:Esther,
                George-Dennis:Frank, Dennis-Esther)

plot(g)

頂点は人を表していて、年齢や性別、関係性について追加したい。

Code 
V(g)$age <- c(25, 31, 18, 23, 47, 22, 50) 
V(g)$gender <- c("f", "m", "f", "m", "m", "f", "m")
E(g)$is_formal <- c(FALSE, FALSE, TRUE, TRUE, TRUE, FALSE, TRUE, FALSE, FALSE)
summary(g)
#> IGRAPH a277a19 UN-- 7 9 -- 
#> + attr: name (v/c), age (v/n), gender (v/c), is_formal (e/l)

ほかのやり方として次がある。

Code 
g <- make_graph(~ Alice-Bob:Claire:Frank, Claire-Alice:Dennis:Frank:Esther,
                George-Dennis:Frank, Dennis-Esther) %>%
  set_vertex_attr("age", value = c(25, 31, 18, 23, 47, 22, 50)) %>%
  set_vertex_attr("gender", value = c("f", "m", "f", "m", "m", "f", "m")) %>%
  set_edge_attr("is_formal", value = c(FALSE, FALSE, TRUE, TRUE, TRUE, FALSE, TRUE, FALSE, FALSE))
summary(g)
#> IGRAPH a293dcc UN-- 7 9 -- 
#> + attr: name (v/c), age (v/n), gender (v/c), is_formal (e/l)

グラフ自体に属性を設定することも可能である。

Code 
g$date <- c("2022-02-11")
graph_attr(g, "date")
#> [1] "2022-02-11"

属性を使ってIDを検索することが可能である。

Code 
V(g)[match(c("George"), V(g)$name)]
#> + 1/7 vertex, named, from a293dcc:
#> [1] George

属性を削除することにする。

Code 
g <- delete_vertex_attr(g, "gender")
V(g)$gender
#> NULL

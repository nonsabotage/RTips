1 はじめに

  • sfはRでGISを扱うためのパッケージです。
  • sfはSimple Featuresの略で、空間データを扱うための標準的なデータモデルです。
  • sfは、Rのtidyverseとの親和性が高く、dplyrのような文法で空間データを扱うことができます。

2 Applied Example

2.1 IDW

データを作成する。

Code 
locations <- 
  tibble(
    longitude = c(10, 10.5, 11),
    latitude = c(10, 14.5, 18),
    measurement = c(100, 150, 120)
  ) |>
  st_as_sf(coords = c("longitude", "latitude"), crs = 4326)

region <- st_as_sf(data.frame(
  x = c(10, 10, 20, 20),
  y = c(10, 20, 20, 10)
), coords = c("x", "y"), crs = 4326)

逆距離補間をおこなう。

grid <- st_make_grid(region, what = "centers")
idw_result <- idw(formula = measurement ~ 1, locations, grid)

3.1 st_rasterize

starsオブジェクトなどをラスターのテンプレートとして、空間データをラスター化する関数です。 leafletでみると少しズレているように見えるけど、QGISでみるとズレていない。

Code 
path_to_polygon <- here(dataset_dir, "meshes_bursa2320.geojson")
polygon <- st_read(path_to_polygon)
#> Reading layer `meshes_bursa2320' from data source 
#>   `C:\Users\suzuk\Dropbox\R\Workspace\RTipsSite\contents\libs\sf\dataset\meshes_bursa2320.geojson' 
#>   using driver `GeoJSON'
#> Simple feature collection with 174928 features and 0 fields
#> Geometry type: POLYGON
#> Dimension:     XY
#> Bounding box:  xmin: 337290.5 ymin: 4383603 xmax: 493290.5 ymax: 4497853
#> Projected CRS: ED50 / TM30
st_write(st_centroid(polygon), here(output_dir, "centroid.geojson"), delete_dsn = TRUE)
#> Deleting source `C:/Users/suzuk/Dropbox/R/Workspace/RTipsSite/contents/libs/sf/output/centroid.geojson' using driver `GeoJSON'
#> Writing layer `centroid' to data source 
#>   `C:/Users/suzuk/Dropbox/R/Workspace/RTipsSite/contents/libs/sf/output/centroid.geojson' using driver `GeoJSON'
#> Writing 174928 features with 0 fields and geometry type Point.
raster_template <- st_as_stars(
  st_bbox(polygon), 
  dx = set_units(250, "m"), 
  dy = set_units(250, "m"))
rasterized <- st_rasterize(polygon, raster_template)
rasterized |> write_stars(here(output_dir, "rasterized.tif"))
rasterized

xx  <- st_transform(st_centroid(head(polygon, 1000)), 4326)
lng <- st_coordinates(xx)[1:1000, 1]
lat <- st_coordinates(xx)[1:1000, 2]
leaflet(
    # options = leafletOptions(crs = leafletCRS(crsClass = "L.CRS.EPSG3857"))
  ) |> 
  addTiles() |>
  addCircleMarkers(lng = lng, lat = lat, radius = .1, color = "blue") |> 
  addPolygons(data = st_transform(head(polygon, 1000), 4326), color = "red") |>  
  addStarsImage(rasterized, colors = "RdPu", opacity = 0.9, project = TRUE)

3.2 st_make_grid

  • 空間データをグリッドに変換する関数です。
  • 以下のように、グリッドのセルの大きさを指定することができます。
Code 
# 例として、ある地域のポリゴンを定義
region <- st_as_sf(data.frame(
  x = c(10, 10, 20, 20),
  y = c(10, 20, 20, 10)
), coords = c("x", "y"), crs = 4326)
Code 
grid <- st_make_grid(region, cellsize = 1, what = "polygons")
grid
#> Geometry set for 100 features 
#> Geometry type: POLYGON
#> Dimension:     XY
#> Bounding box:  xmin: 10 ymin: 10 xmax: 20 ymax: 20
#> Geodetic CRS:  WGS 84
#> First 5 geometries:
#> POLYGON ((10 10, 11 10, 11 11, 10 11, 10 10))
#> POLYGON ((11 10, 12 10, 12 11, 11 11, 11 10))
#> POLYGON ((12 10, 13 10, 13 11, 12 11, 12 10))
#> POLYGON ((13 10, 14 10, 14 11, 13 11, 13 10))
#> POLYGON ((14 10, 15 10, 15 11, 14 11, 14 10))
Code 
plot(grid)
plot(region, add = TRUE)
# ポイントデータを作成することもできる
plot(
  st_make_grid(region, cellsize = 1, what = "centers"),
  col = "red", 
  add = TRUE
)
plot(
  st_make_grid(region, cellsize = 1, what = "corners"),
  col = "blue", 
  add = TRUE
)

セルサイズはXYで変更することもできる。

Code 
plot(grid)
plot(
  st_make_grid(region, cellsize = c(1, 2), what = "centers"),
  col = "red", 
  add = TRUE
)

2 母集団と標本

統計分析には、母集団とその標本、についての解析をおこなう考え方と、データそのものの生成過程について解析を行う方法がある。後者はつまりデータの確率分布を母集団として扱うことになる。

3 練習問題

3.1 2.1

Code 
x <- runif(100)
print(mean(x))
#> [1] 0.4990209
print(var(x))
#> [1] 0.08356039
print(sd(x))
#> [1] 0.2890681
Code 
params <- tibble(
    n_sample =  c(1:1000), 
    n_simulate = 1000
)
result <- 
    params |> 
    mutate(res = map2(n_sample, n_simulate, \(x, y) replicate(y, mean(runif(x))))) |> 
    mutate(mean = map_dbl(res, mean), sd = map_dbl(res, sd))

result |> 
    ggplot(aes(x = n_sample, y = mean)) + 
    geom_line() + 
    scale_x_log10() + 
    scale_y_continuous(
        breaks = 0:10 * .1, 
        limits = c(0, 1)) + 
    geom_ribbon(
        aes(ymin = mean - sd, ymax = mean + sd), fill = "pink", alpha = .5) + 
    theme(
        panel.grid.minor = element_blank()
    )

Code 
library(correlation)

df <- tibble(
    x = runif(100), 
    y = rnorm(100, 0, 1), 
    z = 1.3 * x - .7 * y
)

uv <- 
    combn(syms(c("x", "y", "z")), 2, simplify = FALSE)

graphs <-
    uv |> 
    map(\(x) ggplot(df, aes(!!x[[1]], !!x[[2]])) + geom_point()) |> 
    list_modify(ncol = 1)

do.call(grid.arrange, graphs)

標本サイズと相関係数の関係を確認する。

Code 
library(correlation)

params <- tibble(
    n_sample =  c(10:1000), 
    n_simulate = 100
)
result <- 
    params |> 
    mutate(res1 = map2(
        n_sample, 
        n_simulate, 
        \(x, y) replicate(
            y, {
                xx = runif(x)
                yy = rnorm(x, 0, 1)
                zz = 1.3 * xx - .7 * yy
                c("xy" = cor(xx, yy), "yz" = cor(yy, zz), "zx" = cor(zz, xx))
            }))) |> 
    mutate(res1 = map(res1, \(x) as_tibble(t(x)))) |> 
    mutate(res1 = map(res1, \(x) summarise(x, across(everything(), .fns = list(mean = mean, sd = sd))))) |> 
    unnest(res1)
Code 
cols <- c("xy", "yz", "zx")
graphs <- 
    map(cols, \(x) {
        u <- sym(glue::glue("{x}_mean"))
        v <- sym(glue::glue("{x}_sd"))
        result |> 
            ggplot(aes(x = n_sample, y = !!u)) + 
            geom_path() + 
            geom_ribbon(
                aes(ymin = !!u - !!v, ymax = !!u + !!v), fill = "pink", alpha = .5)
    }) |> 
    list_modify(ncol = 1)

do.call(grid.arrange, graphs)

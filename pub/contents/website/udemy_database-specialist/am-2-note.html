CREATE OR REPLACE TABLE 販売管理.会員
(
    会員コード int not null primary key,
    氏名 varchar(50) not null,
    ポイント int default 0 not null,
    最終使用日 date null
);

CREATE OR REPLACE TABLE 販売管理.商品種別
(
    商品種別コード int not null primary key,
    商品種別名  varchar(100) not null
);


CREATE OR REPLACE TABLE 販売管理.売上
(
    売上番号 int not null primary key,
    売上日時 datetime not null,
    会員コード  int not null,
    利用ポイント int default 0 not null
);

CREATE OR REPLACE TABLE 販売管理.売上明細
(
    売上番号 int not null,
    明細番号 int not null,
    商品コード  int  not null,
    金額 int default 0 not null,
    数量 int default 1 not null,
    primary key (売上番号, 明細番号)
);

CREATE OR REPLACE TABLE 販売管理.商品
(
    商品コード     int not null primary key,
    商品種別コード int not null,
    商品名      varchar(100) not null,
    単価        int  not null,
    最終入荷日  date null,
    型番        varchar(10)  null,
    割引価格    int  null
);

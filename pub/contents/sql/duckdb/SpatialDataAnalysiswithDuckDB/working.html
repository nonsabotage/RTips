Code 
select  
  poi_ist.region as poi_ist_region,
  poi_ist.freeform as poi_ist_freeform,
  poi_ist.categories_main as poi_ist_categori,
  park_ist.categories_main as park_categori, 
  park_ist.names as park_names,
  park_ist.freeform as park_ist_freeform,
  st_distance(poi_ist.geom,park_ist.geom) as dist,
  ST_AsText(poi_ist.geom) as geom,
  ST_AsText(park_ist.geom) as geom2

from poi_ist, park_ist 

where ST_DWithin(poi_ist.geom, park_ist.geom, 500)

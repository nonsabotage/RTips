create or replace table ag_pref_intersection as
select
  COLUMNS(pref.* EXCLUDE(geom)), 
  ag.MESH_ID, 
  ag.SHICODE, 
  ag.PTN_2015,
  st_intersection(pref.geom, ag.geom) as geom
from 
  pref
  inner join ag
  on st_intersects(pref.geom, ag.geom)
;

```{sql connection=con}
create or replace table ag_pref_intersection as
select
  COLUMNS(pref.* EXCLUDE(geom)), 
  ag.MESH_ID, 
  ag.SHICODE, 
  ag.PTN_2015,
  st_intersection(pref.geom, ag.geom) as geom
from 
  pref
  inner join ag
  on st_intersects(pref.geom, ag.geom)
;
```

select

```


どうやらWITH句でランダムを使うと変わってしまうらしい。またマルチポリゴンをポリゴンにキャストする方法は実装されてないようである。

```{sql connection=con}
with 
ta as (
  select 
    gen_random_uuid() as id
    , ST_NGeometries(geom) as ng
    , ROW_NUMBER() OVER() as row
    , geom
  from 
    ag_pref_intersection
  where
    ST_GeometryType(geom) = 'MULTIPOLYGON'
  limit 2
) 
, tb as (
  select 
    id, ng, row
  from 
    ta
  limit 2
)
, ta_tb as (
  select 
    *, 
    unnest(generate_series(1, ta.ng)) as idx
  from
   ta
   inner join tb
   on ta.row = tb.row
)

select 
    *
from
  ta_tb
;


```
